171 bool IsConditionVerified() noexcept
172 {
173 previousComparisonValue = newComparisonValue;
174 switch (comparator)

176 case Comparator::Greater:
177 newComparisonValue = *target > *threshold;
178 break;
179
180 case Comparator::GreaterOrEqual:

182 break;
183
184 case Comparator::Equal:
185 newComparisonValue = *target == *threshold;

187
188 case Comparator::NotEqual:
189 newComparisonValue = *target != *threshold;
190 break;
191
192 case Comparator::Less:
193 newComparisonValue = *target < *threshold;
194 break;
195
196 case Comparator::LessOrEqual:
197 newComparisonValue = *target <= *threshold;
198 break;
199
200 default:
