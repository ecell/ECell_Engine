74 /* Mersenne-Twister (MT-19937) key/state length */
75 #define KEY_LENGTH 624
76
77 /* Internal random number generator state. */
78 typedef struct rng_state_
79 {
80 unsigned long seed; /* integer seed used to initialise state */
81 unsigned long key[KEY_LENGTH]; /* Mersenne-Twister state */
82 int pos; /* current position in key array */
83 int reversed; /* ==0: forward state updates, !=0: reverse state updates */
84 int n_twists; /* number of twists performed */
85 } rng_state;
86
87 /* Initialise generator state from an integer seed. */
88 void init_state(unsigned long seed, rng_state *state);
89
90 /* Optimised implementation of reference Mersenne-Twister from Random Kit. */
91 void twist(rng_state *state);
92
93 /* Reverses twist of state: reverse_twist(twist(state)) is identity map. */
94 void reverse_twist(rng_state *state);
95
96 /* Reverses direction of random number generation. */
97 void reverse(rng_state *state);
98
99 /* Generates a random integer uniformly from range [0, 2^32 - 1]. */
100 unsigned long random_int32(rng_state *state);
101
102 /*
103 * Generate a random double-precision floating point value from uniform
104 * distribution on [0,1).
105 */
106 double random_uniform(rng_state *state);
107
108 /*
109 * Generate a pair of independent random double-precision floating point
110 * values from the (zero-mean, unit variance) standard normal distribution.
111 */
112 void random_normal_pair(rng_state *state, double *ret_1, double *ret_2);
